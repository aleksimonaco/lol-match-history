	VERSION VARCHAR(10) NOT NULL,
	KEY VARCHAR(20) NOT NULL,
	NAME VARCHAR(50) NOT NULL,
	TITLE VARCHAR(50) NOT NULL,
	BLURB VARCHAR(255) NOT NULL,
	INFO_ATTACK INT NOT NULL,
	INFO_DEFENSE INT NOT NULL,
	INFO_MAGIC INT NOT NULL,
	INFO_DIFFICULTY INT NOT NULL,
	IMAGE_FULL VARCHAR(50) NOT NULL,
	IMAGE_GROUP VARCHAR(50) NOT NULL,
	IMAGE_X INT NOT NULL,
	IMAGE_Y INT NOT NULL,
	IMAGE_W INT NOT NULL,
	IMAGE_H INT NOT NULL,
	PARTYPE VARCHAR(50) NOT NULL,
	STATS_HP DECIMAL NOT NULL,
	STATS_HPPERLEVEL DECIMAL NOT NULL,
	STATS_MP DECIMAL NOT NULL,
	STATS_MPPERLEVEL DECIMAL NOT NULL,
	STATS_MOVESPEED DECIMAL NOT NULL,
	STATS_ARMOR DECIMAL NOT NULL,
	STATS_ARMORPERLEVEL DECIMAL NOT NULL,
	STATS_SPELLBLOCK DECIMAL NOT NULL,
	STATS_SPELLBLOCKPERLEVEL DECIMAL NOT NULL,
	STATS_ATTACKRANGE DECIMAL NOT NULL,
	STATS_HPREGEN DECIMAL NOT NULL,
	STATS_HPREGENPERLEVEL DECIMAL NOT NULL,
	STATS_MPREGEN DECIMAL NOT NULL,
	STATS_MPREGENPERLEVEL DECIMAL NOT NULL,
	STATS_CRIT DECIMAL NOT NULL,
	STATS_CRITPERLEVEL DECIMAL NOT NULL,
	STATS_ATTACKDAMAGE DECIMAL NOT NULL,
	STATS_ATTACKDAMAGEPERLEVEL DECIMAL NOT NULL,
	STATS_ATTACKSPEEDOFFSET DECIMAL NOT NULL,
	STATS_ATTACKSPEEDPERLEVEL DECIMAL NOT NULL
);
